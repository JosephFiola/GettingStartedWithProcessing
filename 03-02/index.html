Example 3-2: Draw a Point 

function setup() {
	createCanvas(480, 120);
	background(210);
}

var xPos = 240;
var yPos = 60;

function draw() {
	point(xPos,yPos);
}

function mousePressed(){
	xPos = mouseX;
	yPos = mouseY;
}
