Example 5-21: Move with Arrow Keys

Seems the function keyPressed() needs to be present for keyCode detection to work
Use keyCode = LEFT_ARROW instead of keyCode = LEFT
var x = 215;

function draw() {
  background(200);
  if (keyIsPressed){
    if (keyCode == LEFT_ARROW){
      x--;
    } else if (keyCode == RIGHT_ARROW){
      x++;
    }
  }
rect(x, 45, 50, 50);
}

function keyPressed() {
  return false; // prevent default
}
