Example 5-2: The setup() Function

Open your browser's console window to see the output of the print() function.
This example is also using the text() function to print directly into the canvas. read more
Also added some functionality to update the text display on the canvas. 
var textPosition = 0;
var startUpText;
	
function setup() {
  startUpText = "I'm starting";
  print(startUpText);        
  text("I'm starting", 20, 20, 600, 80);
}

function draw() {
  textPosition++;
  if (textPosition>24)
  {	
    background(200);
    text(startUpText, 20, 20, 600, 80);
    textPosition = 1;
    }
  print("I'm running (frame " + frameCount + ")");
  text ("I'm running (frame " + frameCount + ")", 20, 20+(textPosition*15), 600, 80);
}
