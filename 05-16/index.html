Example 5-16: The Bounds of a Circle

var x = 120;
var y = 60;
var radius = 12;

function draw() {
  var d = dist(mouseX, mouseY, x, y);
  if (d < radius) {
    radius++;
    fill(0);
  } else {
    fill(255);
  }
  ellipse(x, y, radius, radius);
}
