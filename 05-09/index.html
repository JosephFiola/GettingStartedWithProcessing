Example 5-9: Smooth Lines with Easing

var x = 0;
var y = 0;
var px = 0;
var py = 0;
var easing = 0.05;

function draw() {
  var targetX = mouseX;
  x += (targetX - x) * easing;
  var targetY = mouseY;
  y += (targetY - y) * easing;
  var weight = dist(x, y, px, py);
  strokeWeight(weight);
  line(x, y, px, py);
  py = y;
  px = x;
}
